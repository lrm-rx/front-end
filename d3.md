npm install vue d3

## 创建 Vue 组件
```vue
<template>
  <div ref="chartContainer" class="chart-container">
    <!-- 图表将渲染在这里 -->
  </div>
</template>

<script setup>
import { onMounted, ref, watch } from 'vue';
import * as d3 from 'd3';

const chartContainer = ref(null);

// 数据示例
const data = {
  nodes: [
    { id: 'A', x: 100, y: 100, label: '节点 A' },
    { id: 'B', x: 200, y: 100, label: '节点 B' },
    { id: 'C', x: 150, y: 200, label: '节点 C' }
  ],
  links: [
    { source: 'A', target: 'B' },
    { source: 'B', target: 'C' }
  ]
};

onMounted(() => {
  const svg = d3.select(chartContainer.value).append('svg')
    .attr('width', '100%')
    .attr('height', '100%');

  const width = chartContainer.value.clientWidth;
  const height = chartContainer.value.clientHeight;

  svg.attr('viewBox', `0 0 ${width} ${height}`)
     .attr('preserveAspectRatio', 'xMinYMin meet');

  // 创建链接
  const link = svg.append('g').selectAll('.link')
    .data(data.links)
    .enter().append('line')
    .attr('class', 'link')
    .attr('x1', d => data.nodes.find(node => node.id === d.source).x)
    .attr('y1', d => data.nodes.find(node => node.id === d.source).y)
    .attr('x2', d => data.nodes.find(node => node.id === d.target).x)
    .attr('y2', d => data.nodes.find(node => node.id === d.target).y)
    .style('stroke', '#999')
    .style('stroke-width', '2px');

  // 创建节点
  const node = svg.append('g').selectAll('.node')
    .data(data.nodes)
    .enter().append('g')
    .attr('class', 'node');

  node.append('circle')
    .attr('r', 20)
    .attr('cx', d => d.x)
    .attr('cy', d => d.y)
    .style('fill', '#69b3a2');

  node.append('text')
    .attr('x', d => d.x)
    .attr('y', d => d.y)
    .attr('dy', '.35em')
    .attr('text-anchor', 'middle')
    .text(d => d.label)
    .style('fill', '#fff');
});

watch(() => chartContainer.value.clientWidth, (newWidth) => {
  const svg = d3.select(chartContainer.value).select('svg');
  svg.attr('width', newWidth);
  svg.attr('viewBox', `0 0 ${newWidth} ${chartContainer.value.clientHeight}`);
});
</script>

<style scoped>
.chart-container {
  width: 100%;
  height: 100%;
  position: relative;
}
.node circle {
  stroke: #fff;
  stroke-width: 1.5px;
}
.link {
  stroke: #999;
  stroke-width: 2px;
}
</style>
```
